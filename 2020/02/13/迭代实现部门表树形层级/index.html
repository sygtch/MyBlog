# 迭代实现部门表树形层级

方案1：先在数据库查出一级部门，再遍历每个部门，用其id作为父id查下属部门

public void getDept(){
    //取出所有部门,返回的即是树形结构
    List<Department> list = this.getEvery(0);

}

//传入参数时，默认这时是零级部门，委托该方法用自身id=0去查找子部门
public List<Department> getEvery(int pid){

    //将入参作为父id，从数据库查询子部门。此时，next为一级部门
    List<Department> next = deptRepository.findByParentId(pid);

    if (next.size() > 0){
        //如果当前所委托查找的pid，确实存在子部门
        for (Department od : next) {
            //对子部门遍历，继续查找下属部门
            List<Department> sub = getEvery(od.getDept_id());
            //一切查完后，如果当前子部门确实存在子部门（孙子部门）。此时，sub为二级部门。
            //将孙子部门设置为od下属
            if (null != sub && sub.size() > 0){
                od.setDepartment(sub);
            }
        }
        return next;
    }

    return null;
}

方案2：一次性查出所有部门，减少数据库操作

@Service
public class DepartmentServiceImpl extends ServiceImpl<DepartmentMapper, Department> implements DepartmentService {

    @Autowired
    private DepartmentMapper departmentMapper;

    //所有部门
    private List<Department> allDepts;

    @Override
    public List<Department> getMenus() {
        //查出department表的所有字段，避免多次查询的多次连接数据库
        allDepts = departmentMapper.selectList(new QueryWrapper<>());
        //查父级id是0的所有department
        return getSub(0);
    }

    //用pid查询其下属部门
    private List<Department> getSub(Integer pid){
        //在所有部门中筛选parentId符合条件的项，组成list
        //这里的depts和for循环只是为了避免多次连接数据库，所以在此筛选
        List<Department> depts = new ArrayList<>();
        for (Department dept : allDepts) {
            if (pid.equals(dept.getParentId())){
                depts.add(dept);
            }
        }

        //对符合条件的 depts 进行遍历
        for (Department dept : depts) {
            //得到其下属部门 subs
            List<Department> subs = getSub(dept.getDeptId());
            if (null != subs && subs.size()>0){
                //将下属部门添加到当前 dept 部门
                dept.setDepartments(subs);
            }
        }
        return depts;
    }
}

    * 部门名称

     */
