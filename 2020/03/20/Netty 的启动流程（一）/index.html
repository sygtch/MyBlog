# Netty 的客户端启动流程（一）

客户端

/**
 * @author GuoTianchi

 * @date 2020/4/11 14:27

public class ClientProxy {

    private Object connect(RpcProtocol protocol){
        //1.初始化
        Bootstrap bootstrap = new Bootstrap();
        //2.自定义执行器
        final ClientHandler handler = new ClientHandler();
        //3.work“线程工作组”
        EventLoopGroup work = new NioEventLoopGroup();
        //4.设置group
        bootstrap.group(work)
                 //5.设置channel
                .channel(NioSocketChannel.class)
                .option(ChannelOption.TCP_NODELAY,true)
                 //6.配置handler
                .handler(new ChannelInitializer<SocketChannel>() {
                    @Override
                    protected void initChannel(SocketChannel ch) throws Exception {
                        ChannelPipeline pipeline = ch.pipeline();
                        pipeline.addLast("frameDecoder", new LengthFieldBasedFrameDecoder(Integer.MAX_VALUE, 0, 4, 0, 4));
                        //自定义协议编码器
                        pipeline.addLast("frameEncoder", new LengthFieldPrepender(4));
                        //对象参数类型编码器
                        pipeline.addLast("encoder", new ObjectEncoder());
                        //对象参数类型解码器
                        pipeline.addLast("decoder", new ObjectDecoder(Integer.MAX_VALUE, ClassResolvers.cacheDisabled(null)));
                        pipeline.addLast("handler",handler);
                    }
                });
        try {
            //7.
            ChannelFuture future = bootstrap.connect("localhost", 8080).sync();
            Channel channel = future.channel();
            channel.writeAndFlush(protocol).sync();
            channel.closeFuture().sync();
        } catch (InterruptedException e) {
            e.printStackTrace();
        } finally {
            work.shutdownGracefully();
        }
        return handler.getResponse();
    }
}

  1. Bootstrap主要完成Netty的初始化；

  2. 自定义的一个Handler类，完成自己的业务逻辑；

  3. 相当于一个线程池，里面存有一个线程工作组，如果初始化时不传值，则默认为CPU内核数的二倍；

  4. 设置group；

  5. 设置channel：

        public B channel(Class<? extends C> channelClass) {
        if (channelClass == null) {
            throw new NullPointerException("channelClass");
        }
        return channelFactory(new ReflectiveChannelFactory<C>(channelClass));
    }

    先看ReflectiveChannelFactory<C>(channelClass)做了什么，实际上就是把NioSocketChannel的class存起来，完成channelFactory的初始化。

        public ReflectiveChannelFactory(Class<? extends T> clazz) {
        if (clazz == null) {
            throw new NullPointerException("clazz");
        }
        this.clazz = clazz;
    }

    然后再看channelFactory()方法，实际就是把刚刚初始化的reflectiveChannelFactory实例存起来：

        public B channelFactory(ChannelFactory<? extends C> channelFactory) {
        ...     
        this.channelFactory = channelFactory;

  1. 设置handler；

  2. 开始连接，初始化入口：

        public ChannelFuture connect(String inetHost, int inetPort) {
        return connect(InetSocketAddress.createUnresolved(inetHost, inetPort));
    }

    进入connect()方法：

        public ChannelFuture connect(SocketAddress remoteAddress) {
        if (remoteAddress == null) {
            throw new NullPointerException("remoteAddress");
        }
        validate();
        //注意这里
        return doResolveAndConnect(remoteAddress, config.localAddress());
    }

    进入doResolveAndConnect()方法：

    private ChannelFuture doResolveAndConnect(final SocketAddress remoteAddress, final SocketAddress localAddress) {
    //7.1 初始化、注册
    final ChannelFuture regFuture = initAndRegister();
    final Channel channel = regFuture.channel();
    if (regFuture.isDone()) {
        if (!regFuture.isSuccess()) {
            return regFuture;
        }
        return doResolveAndConnect0(channel, remoteAddress, localAddress, channel.newPromise());
    } else {
        final PendingRegistrationPromise promise = new PendingRegistrationPromise(channel);
        regFuture.addListener(new ChannelFutureListener() {
            @Override
            public void operationComplete(ChannelFuture future) throws Exception {

                Throwable cause = future.cause();
                if (cause != null) {
                    promise.setFailure(cause);
                } else {
                    promise.registered();
                    doResolveAndConnect0(channel, remoteAddress, localAddress, promise);
                }
            }
        });
        return promise;
    }
}

    7.1 初始化、注册：

    final ChannelFuture initAndRegister() {
    Channel channel = null;
    try {
        //7.1.1利用之前储存的reflectiveChannelFactory实例和NioSocketChannel.class初始化一个nioSocketChannel实例
        channel = channelFactory.newChannel();
        //将bootstrap的各种配置加载进去
        init(channel);
    } catch (Throwable t) {
        if (channel != null) {
            channel.unsafe().closeForcibly();
        }
        return new DefaultChannelPromise(channel, GlobalEventExecutor.INSTANCE).setFailure(t);
    }
    //这里的group就是开始的group(work)
    //7.1.2 register方法
    ChannelFuture regFuture = config().group().register(channel);
    if (regFuture.cause() != null) {
        if (channel.isRegistered()) {
            channel.close();
        } else {
            channel.unsafe().closeForcibly();
        }
     }
    return regFuture;
}

    7.1.1 channelFactory.newChannel()方法，调用NioSocketChannel()默认构造。涉及unsafe初始化，pipeline初始化：

    public NioSocketChannel() {

        this(DEFAULT_SELECTOR_PROVIDER);
}

public NioSocketChannel(SelectorProvider provider) {

    this(newSocket(provider));
}

public NioSocketChannel(Channel parent, SocketChannel socket) {
    super(parent, socket);
    config = new NioSocketChannelConfig(this, socket.socket());
}

//查看super方法，io.netty.channel.nio.AbstractNioByteChannel
protected AbstractNioByteChannel(Channel parent, SelectableChannel ch) {
    //加入Selector选择器的SelectionKey.OP_READ
    super(parent, ch, SelectionKey.OP_READ);
}

//继续进入super方法，io.netty.channel.nio.AbstractNioChannel
protected AbstractNioChannel(Channel parent, SelectableChannel ch, int readInterestOp) {
    super(parent);
    //这里要记住，一会取ch，这个ch是Java Nio中的SocketChannel
    this.ch = ch;
    this.readInterestOp = readInterestOp;
    try {
        //设置为非阻塞

    } catch (IOException e) {

            ch.close();
        } catch (IOException e2) {
            if (logger.isWarnEnabled()) {
                logger.warn(
                    "Failed to close a partially initialized socket.", e2);
            }
        }
        throw new ChannelException("Failed to enter non-blocking mode.", e);
    }
}

    this.parent = parent;
    id = newId();

    unsafe = newUnsafe();

    7.1.2 register()方法：

    public ChannelFuture register(Channel channel) {
    //将work工作组中的一个线程注册
    return next().register(channel);
}

public ChannelFuture register(Channel channel) {
    return register(new DefaultChannelPromise(channel, this));
}

public ChannelFuture register(final ChannelPromise promise) {
    ObjectUtil.checkNotNull(promise, "promise");
    //这里的channel还是上面的nioSocketChannel实例
    //返回一个unsafe字段
    //进入register方法
    promise.channel().unsafe().register(this, promise);
    return promise;
}

再次进入register()方法，这里实际上就是把eventLoop（工作组中的一个线程）和客户端传来的channel关联起来：

   //这里的eventLoop是刚刚next()取出来的单个线程，promise是包装了nioSocketChannel的实例
   public final void register(EventLoop eventLoop, final ChannelPromise promise) {

       ...

       //把eventLoop赋值，其实这里就是之后next()取出的
       AbstractChannel.this.eventLoop = eventLoop;

       if (eventLoop.inEventLoop()) {
           //进入register0方法
           register0(promise);
       } else {
           try {
               eventLoop.execute(new Runnable() {
                   @Override
                   public void run() {
                       register0(promise);
                   }
               });
           } catch (Throwable t) {
               logger.warn(
                   "Force-closing a channel whose registration task was not accepted by an event loop: {}",
                   AbstractChannel.this, t);
               closeForcibly();
               closeFuture.setClosed();
               safeSetFailure(promise, t);
           }
       }
   }

进入register0()方法：

   private void register0(ChannelPromise promise) {
       try {
           ...
           boolean firstRegistration = neverRegistered;
           //进入doRegister方法
           doRegister();
           neverRegistered = false;
           registered = true;
           pipeline.invokeHandlerAddedIfNeeded();
           safeSetSuccess(promise);
           pipeline.fireChannelRegistered();

           if (isActive()) {
               if (firstRegistration) {
                   pipeline.fireChannelActive();
               } else if (config().isAutoRead()) {
                   beginRead();
               }
           }
       } catch (Throwable t) {
           closeForcibly();
           closeFuture.setClosed();
           safeSetFailure(promise, t);
       }
   }

进入doRegister()方法，用java Nio的SocketChannel对选择器进行注册（nio底层的原生方法）：

   protected void doRegister() throws Exception {
       boolean selected = false;
       for (;;) {
           try {
               //用java Nio的SocketChannel对选择器进行注册，nio底层的原生方法
               selectionKey = javaChannel().register(eventLoop().selector, 0, this);
               return;
           } catch (CancelledKeyException e) {
               if (!selected) {
                   eventLoop().selectNow();
                   selected = true;
               } else {
                   throw e;
               }
           }
       }
   }

这时，再回到bootstrap的connect()中doConnect()的方法，在该方法中的channel是实例nioSocketChannel（上面eventLoop和nioSocketChannel时分析的）：

   private static void doConnect(
       final SocketAddress remoteAddress, final SocketAddress localAddress, final ChannelPromise connectPromise) {
       final Channel channel = connectPromise.channel();
       //每个channel都已经和一个eventLoop关联
       channel.eventLoop().execute(new Runnable() {
           @Override
           public void run() {
               if (localAddress == null) {
                   //进入connect方法
                   channel.connect(remoteAddress, connectPromise);
               } else {
                   channel.connect(remoteAddress, localAddress, connectPromise);
               }
               connectPromise.addListener(ChannelFutureListener.CLOSE_ON_FAILURE);
           }
       });
   }

进入connect()方法：

   public final ChannelFuture connect(
       SocketAddress remoteAddress, SocketAddress localAddress, ChannelPromise promise) {
       return tail.connect(remoteAddress, localAddress, promise);
   }

tail,其实它是一个TailContext 的实例，而TailContext 又是AbstractChannelHandlerContext 的子类，但是TailContext 没有实现 connect()方法，因此这里调用的其实是 AbstractChannelHandlerContext的connect()方法：

   public ChannelFuture connect(
       final SocketAddress remoteAddress, final SocketAddress localAddress, final ChannelPromise promise) {

       ...

       //这里是关键的地方
       final AbstractChannelHandlerContext next = findContextOutbound();
       EventExecutor executor = next.executor();
       if (executor.inEventLoop()) {
           //调用head的方法进行连接
           next.invokeConnect(remoteAddress, localAddress, promise);
       } else {
           safeExecute(executor, new Runnable() {
               @Override
               public void run() {
                   next.invokeConnect(remoteAddress, localAddress, promise);
               }
           }, promise, null);
       }
       return promise;
   }

这里调用findContextOutbound()方法，从DefaultChannelPipeline内的双向链表的tail开始，一直向前，找到第一个outbound为true的AbstractChannelHandlerContext，然后调用它的invokeConnect()方法：

   private AbstractChannelHandlerContext findContextOutbound() {
       AbstractChannelHandlerContext ctx = this;
       do {
           ctx = ctx.prev;
       } while (!ctx.outbound);
       return ctx;
   }

head 是 HeadContext的实例，它实现了ChannelOutboundHandler接口，并且它的outbound设置为true。因此在findContextOutbound()方法中，找到的AbstractChannelHandlerContext对象其实就是head。之后再回到next.invokeConnect()方法：

   private void invokeConnect(SocketAddress remoteAddress, SocketAddress localAddress, ChannelPromise promise) {
       if (invokeHandler()) {
           try {
               //handler()返回的就是this，即head节点。
               //而head实现了ChannelOutboundHandler接口，所以可以向上转型
               //且head节点是HeadContext的实例，该类中重写了connect()方法
               ((ChannelOutboundHandler) handler()).connect(this, remoteAddress, localAddress, promise);
           } catch (Throwable t) {
               notifyOutboundHandlerException(t, promise);
           }
       } else {
           connect(remoteAddress, localAddress, promise);
       }
}

而又因为 HeadContext重写了connect()方法，因此实际上调用的是 HeadContext的connect()方法：

   public void connect(
       ChannelHandlerContext ctx,
       SocketAddress remoteAddress, SocketAddress localAddress,
       ChannelPromise promise) throws Exception {
       unsafe.connect(remoteAddress, localAddress, promise);
}

查看HeadContext的构造器：

   HeadContext(DefaultChannelPipeline pipeline) {
       super(pipeline, null, HEAD_NAME, false, true);
       //pipeline.channel()直接返回channel
       //unsafe返回nioSocketChannelUnsafe实例
    unsafe = pipeline.channel().unsafe();
       setAddComplete();
}

我们知道，此时的unsafe其实就是nioSocketChannel的unsafe字段，即AbstractChannel中的unsafe字段，即nioSocketChannelUnsafe实例，所以继续进入unsafe.connect()方法。但在NioSocketChannelUnsafe中没有重写connect方法，所以走父类AbstractNioUnsafe的connect方法：

   public final void connect(
       final SocketAddress remoteAddress, final SocketAddress localAddress, final ChannelPromise promise) {
              ...
           boolean wasActive = isActive();
           //这里调用的是AbstractNioChannel的抽象方法。
           //doConnect()方法是在 NioSocketChannel中实现的
           if (doConnect(remoteAddress, localAddress)) {
               fulfillConnectPromise(promise, wasActive);
        } 
           ...
}

doConnect()方法在该类中是抽象的，所以按照最初的实例初始化顺序去找，找到其子类NioSocketChannel重写了该方法，因此进入到NioSocketChannel的doConnect()方法中：

   protected boolean doConnect(SocketAddress remoteAddress, SocketAddress localAddress) throws Exception {
       if (localAddress != null) {
           doBind0(localAddress);
       }

       boolean success = false;
       try {
           //调用java的Channel，用底层方法进行连接
           //这里的javaChannel()和以前的一样，还是返回ch
           boolean connected = javaChannel().connect(remoteAddress);
           if (!connected) {
               selectionKey().interestOps(SelectionKey.OP_CONNECT);
           }
           success = true;
           return connected;
       } finally {
           if (!success) {
               doClose();
        }
       }
}

至此，客户端的连接已经基本完成了…

服务端

在这里需要注意的是，客户端的 unsafe 是 AbstractNioByteChannel#NioByteUnsafe 的实例，而服务端的unsafe是AbstractNioMessageChannel.AbstractNioUnsafe 的实例。因为AbstractNioMessageChannel 重写了 newUnsafe()方法。

其实一开始我有点看晕了，搞不明白为什么服务端这里的newUnsafe()方法走到哪了！！!后来发现，其实NioServerSocketChannel在初始化的路上，走到了AbstractNioMessageChannel 时，重写了newUnsafe()方法；

而同样的，NioSocketChannel在初始化的一路上，走到了AbstractNioByteChannel时，重写了newUnsafe()方法；
